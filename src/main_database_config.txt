import { Sequelize } from 'sequelize';
import path from 'path';
import { fileURLToPath } from 'url';
import 'sqlite3'; // Static import to ensure Vercel bundles the dialect
import 'pg';
import 'pg-hstore';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

let sequelize;

if (process.env.DATABASE_URL) {
  // Production: PostgreSQL (Vercel / Render with Postgres)
  console.log('­ƒöÁ Database: Using PostgreSQL (DATABASE_URL)');
  sequelize = new Sequelize(process.env.DATABASE_URL, {
    dialect: 'postgres',
    dialectOptions: {
      ssl: {
        require: true,
        rejectUnauthorized: false,
      },
    },
    logging: false,
  });
} else {
  // Local Dev: SQLite
  const storagePath = process.env.VERCEL ? '/tmp/database.sqlite' : path.join(__dirname, '../database.sqlite');
  console.log('­ƒùä´©Å  Database: Initializing local SQLite (path: ' + storagePath + ')');
  sequelize = new Sequelize({
    dialect: 'sqlite',
    storage: storagePath,
    logging: false,
  });
}

export default sequelize;
